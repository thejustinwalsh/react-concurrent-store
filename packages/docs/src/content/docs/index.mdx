```tsx
import { createStore, useStore } from "react-use-store";
import { Suspense, use, useCallback, useTransition } from "react";

type User = { id: number; name: string; email: string };

// Create a store that manages an async resource
const fetchUser = async (id: number): Promise<User> => {
  const response = await fetch(
    `https://jsonplaceholder.typicode.com/users/${id}`
  );
  return response.json();
};

const userStore = createStore(fetchUser(1));

function UserProfile() {
  const userPromise = useStore(userStore);
  const user = use(userPromise);

  const [isPending, startTransition] = useTransition();
  const handleLoadNextUser = useCallback(() => {
    startTransition(() => {
      userStore.update(fetchUser(user.id + 1));
    });
  }, [user.id]);

  return (
    <div>
      <div style={{ opacity: isPending ? 0.5 : 1 }}>
        <h1>{user.name}</h1>
        <p>Email: {user.email}</p>
      </div>
      <button onClick={handleLoadNextUser}>Load Next User</button>
    </div>
  );
}

function App() {
  return (
    <Suspense fallback={<div>Loading...</div>}>
      <UserProfile />
    </Suspense>
  );
}
```
